	public static void calificacionUsario(){
		String strCalif = jOptionPane.showInputDialog("Escribe tu calificación");

		if (calificacion < 70) {
			System.out.println("Reprobado");

	}

	public static void caloriasConsumidas(){
		String strComidas = jOptionPane.showInputDialog("Escribe tu número de comidas");
		int numeroComidas = Integer.parseInt(strComidas);
		int totalCalorias = 0;
		for(int i = 0; i<=numeroComidas;i++){
			String strCalorias = jOptionPane.showInputDialog("Escribe las calorías que consumiste en tu comida #" + i);
			int calorias = Integer.parseInt(strCalorias);
			totalCalorias += calorias;
		}
		System.out.println("Consumiste un total de " + totalCalorias + "calorias.")
	}
